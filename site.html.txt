  <script>
    // Inicializa o visualizador
    const viewer = new PhotoSphereViewer.Viewer({
      container: document.getElementById('viewer'),
      panorama: 'imagem1.jpg', // primeira imagem que abre
      caption: 'Galeria 360°',
      loadingImg: 'https://photo-sphere-viewer.js.org/assets/photosphere-logo.gif',
      touchmoveTwoFingers: true,
      mousewheelCtrlKey: false, // permite zoom direto no scroll
      navbar: [], // esconde barra padrão
      minFov: 30, // limite de zoom máximo
      maxFov: 100 // limite de zoom mínimo
    });

    // Trocar imagem e destacar miniatura selecionada
    function trocarImagem(elemento, nomeArquivo) {
      viewer.setPanorama(nomeArquivo);
      document.querySelectorAll('#galeria img').forEach(img => img.classList.remove('selecionada'));
      elemento.classList.add('selecionada');
    }

    // Ativar/desativar fullscreen
    function toggleFullscreen() {
      if (!document.fullscreenElement) {
        document.documentElement.requestFullscreen();
      } else {
        if (document.exitFullscreen) {
          document.exitFullscreen();
        }
      }
    }

    // Funções de zoom
    function zoomIn() {
      let fov = viewer.getZoomLevel();
      if (fov > 30) viewer.zoom(fov - 10); // aproxima
    }
    function zoomOut() {
      let fov = viewer.getZoomLevel();
      if (fov < 100) viewer.zoom(fov + 10); // afasta
    }
  </script>
